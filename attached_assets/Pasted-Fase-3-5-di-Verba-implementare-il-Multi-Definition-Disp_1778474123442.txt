per parole polisemiche (Magoosh-style).

# CONTESTO
- quizQueries.ts già ritorna QuizWord.allDefinitions: array di { part_of_speech, 
  definition, example, display_order } già ordinato per display_order.
- Per parole monosemiche (1 def) FeedbackCard mostra già correctDefinition + 
  exampleSentence — questo comportamento DEVE rimanere invariato.
- Per parole polisemiche (2-3 def) vogliamo mostrare TUTTE le definizioni come

- Sample DB con multi-def: la parola "sanguine" ha 3 definizioni (adjective: 
  optimistic / adjective: ruddy / noun: blood-red colour).

# STEP 2 — CREA SUB-COMPONENT FeedbackMultiDefinitions
Nello stesso file FeedbackCard.tsx, sotto FeedbackExample, aggiungi:

interface FeedbackMultiDefinitionsProps {
  definitions: { part_of_speech: string; definition: string; example: string; display_order: number }[];
  visible?: boolean;
}
export function FeedbackMultiDefinitions({ definitions, visible = true }: FeedbackMultiDefinitionsProps) {
  if (!visible || !definitions || definitions.length === 0) return null;
  
  return (
    <div style={{ marginTop: 16, display: "flex", flexDirection: "column", gap: 18 }}>
      {definitions.map((def, idx) => (
        <div key={idx} style={{ 
          paddingLeft: idx === 0 ? 0 : 14, 
          borderLeft: idx === 0 ? "none" : "1px solid rgba(255,255,255,0.08)",
        }}>
          {/* Part-of-speech tag */}
          <p style={{
            fontFamily: "'Inter', sans-serif",
            fontWeight: 500,
            fontSize: 11,
            letterSpacing: "0.12em",
            textTransform: "lowercase",
            color: "rgba(199,184,232,0.5)",
            margin: "0 0 6px",
            fontStyle: "italic",
          }}>
            {def.part_of_speech}
          </p>
          {/* Definition */}
          <p style={{
            fontFamily: "'Inter', sans-serif",
            fontWeight: 400,
            fontSize: idx === 0 ? 20 : 17,
            color: "#FFFFFF",
            margin: 0,
            lineHeight: 1.4,
          }}>
            {def.definition}
          </p>
          {/* Example */}
          {def.example && (
            <p style={{
              fontFamily: "'Inter', sans-serif",
              fontWeight: 300,
              fontSize: 15,
              fontStyle: "italic",
              color: "rgba(255,255,255,0.6)",
              margin: "8px 0 0",
              lineHeight: 1.5,
            }}>
              "{def.example}"
            </p>
          )}
        </div>
      ))}
    </div>
  );
}
